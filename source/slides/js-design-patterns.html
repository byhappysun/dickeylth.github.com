单例模式

生活中的单例模式

主要特点

  • 单个实例
  • 初始化
  • 对外提供访问接口

实现方式

  • 对象字面量
  • 使用new创建对象

对象字面量

                        
var x = {
    attr: 'value'
};

var y = {
    attr: 'value'
};

x == y;     // false
x === y;    // false

加强版

                        
var SuperMario = (function(){

    var instance = null;

    // 初始化函数
    function init(){

        var age = 12,
            height = 120;

        // 门面模式返回成员属性
        return {

            name: 'Mario',

            getAge: function(){
                return age;
            },
            getHeight: function(){
                return height;
            },

            jump: function(){
                console.log("I'm jumping!");
            },
            run: function(){
                console.log("I'm running!");
            }
        };
    }

    return {

        // 获取实例，懒加载
        getInstance: function(){
            if(!instance){
                instance = init();
            }
            return instance;
        }

    };

})();
console.log(SuperMario.getInstance());
